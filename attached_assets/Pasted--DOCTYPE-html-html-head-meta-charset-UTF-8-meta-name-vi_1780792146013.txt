    const navButtons = ['nav-dashboard', 'nav-workout', 'nav-calorie', 'nav-water', 'nav-profile'];
    navButtons.forEach(id => {
        const btn = document.getElementById(id);
        if (!btn) return;
        btn.addEventListener('click', function() {
            navButtons.forEach(nid => {
                const nb = document.getElementById(nid);
                if (!nb) return;
                nb.style.background = 'transparent';
                nb.style.border = 'none';
                nb.querySelectorAll('i').forEach(i => {
                    i.style.color = 'rgba(247,235,232,0.4)';
                });
                nb.querySelectorAll('span').forEach(s => {
                    s.style.color = 'rgba(247,235,232,0.4)';
                    s.style.fontWeight = '300';
                });
            });
            this.style.background = 'rgba(201,137,122,0.2)';
            this.style.border = '1px solid rgba(201,137,122,0.3)';
            this.querySelectorAll('i').forEach(i => {
                i.style.color = '#C9897A';
            });
            this.querySelectorAll('span').forEach(s => {
                s.style.color = '#C9897A';
                s.style.fontWeight = '500';
            });
        });
    });

    const waterBtns = document.querySelectorAll('#hydration-card button:not([class*="justify-between"])');
    const addWaterAmounts = [250, 500, 750];
    let currentWater = 1600;
    const waterGoal = 2500;

    document.querySelectorAll('#hydration-card .flex.gap-2.mt-3 button').forEach((btn, idx) => {
        btn.addEventListener('click', function() {
            currentWater = Math.min(currentWater + addWaterAmounts[idx], waterGoal);
            const pct = Math.round((currentWater / waterGoal) * 100);
            const liters = (currentWater / 1000).toFixed(1);
            const remaining = ((waterGoal - currentWater) / 1000).toFixed(1);
            const dashoffset = 175.9 - (175.9 * pct / 100);

            const ring = document.querySelector('#hydration-card circle:nth-child(2)');
            if (ring) ring.setAttribute('stroke-dashoffset', dashoffset.toString());

            const pctEl = document.querySelector('#hydration-card .absolute.inset-0.flex span');
            if (pctEl) pctEl.textContent = pct + '%';

            const rows = document.querySelectorAll('#hydration-card .flex-1.flex.flex-col.gap-2 .flex.items-center.justify-between span:last-child');
            if (rows[0]) rows[0].textContent = liters + ' L';
            if (rows[1]) rows[1].textContent = remaining + ' L';
        });
    });
</script>
